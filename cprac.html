8 write algoalgorithm and code calculate roots of quadratic equation using if else in c language

Algorithm

  1. Input the coefficients of the quadratic equation ax^2 + bx + c = 0.
  2. Calculate the discriminant D = b^2 - 4ac
  3. If D > 0, there are two real roots.
  4. Calculate the two roots using the formula r1 = (-b + sqrt(D))/2a and r2 = (-b - sqrt(D))/2a.
  5. If D = 0, there is only one real root. Calculate the root using the formula r = -b/2a.
  6. If D < 0, there are two complex conjugate roots.
  7. Calculate the two complex conjugate roots using the formula r1 = (-b + sqrt(D))/2a and r2 = (-b - sqrt(D))/2a.

CODE 
                    #include < stdio.h>
                        #include < math.h>
                        
                        int main() 
                        {
                            float a, b, c, D, r1, r2;
                            
                            // Input the coefficients of the quadratic equation
                            printf("Enter the values of a, b and c: ");
                            scanf("%f %f %f", &a, &b, &c);
                            
                            // Calculate the discriminant
                            D = b*b - 4*a*c;
                            
                            // If discriminant is greater than 0, there are two real roots
                            if (D > 0)
                            {
                                r1 = (-b + sqrt(D))/(2*a);
                                r2 = (-b - sqrt(D))/(2*a);
                                printf("The roots are %.2f and %.2f.", r1, r2);
                            }
                            
                            // If discriminant is equal to 0, there is one real root
                            else if (D == 0)
                            {
                                r1 = r2 = -b/(2*a);
                                printf("The root is %.2f.", r1);
                            }
                            
                            // If discriminant is less than 0, there are two complex conjugate roots
                            else
                            {
                                r1 = r2 = -b/(2*a);
                                printf("The roots are %.2f+i%.2f and %.2f-i%.2f.", r1, sqrt(-D)/(2*a), r2, sqrt(-D)/(2*a));
                            }
                            
                            return 0;
                        }

16. write algoalgorithm and code to take input perform matrix multiplication by checking the compaatibility in c language

Algorithm

  1. Take two matrices of order m x n and n x p as input.
  2. Check the compatibility of the matrices for multiplication.
  3. If the matrices are compatible then perform the multiplication.
  4. Initialize a result matrix of order m x p.
  5. Compute the sum of the product of the corresponding row and column elements of the given matrices and store the result in the result matrix.
  6. Display the result matrix.

CODE 
                    #include < stdio.h>

                        int main()
                        {
                            int m, n, p, i, j, k;
                            int first[10][10], second[10][10], multiply[10][10];
                        
                            // Take input for two matrices
                            printf("Enter the number of rows and columns of first matrix: ");
                            scanf("%d %d", &m, &n);
                            printf("Enter the number of rows and columns of second matrix: ");
                            scanf("%d %d", &n, &p);
                        
                            // Check the compatibility of the given matrices for multiplication
                            if (n != p)
                            {
                                printf("Matrices with entered orders can't be multiplied with each other.\n");
                            }
                            else
                            {
                                printf("Enter elements of first matrix: \n");
                                for (i = 0; i < m; ++i)
                                {
                                    for (j = 0; j < n; ++j)
                                    {
                                        scanf("%d", &first[i][j]);
                                    }
                                }
                        
                                printf("Enter elements of second matrix: \n");
                                for (i = 0; i < n; ++i)
                                {
                                    for (j = 0; j < p; ++j)
                                    {
                                        scanf("%d", &second[i][j]);
                                    }
                                }
                        
                                // Initialize the result matrix with 0
                                for (i = 0; i < m; ++i)
                                {
                                    for (j = 0; j < p; ++j)
                                    {
                                        multiply[i][j] = 0;
                                    }
                                }
                        
                                // Compute the sum of the product of the corresponding row and column elements of the given matrices
                                for (i = 0; i < m; ++i)
                                {
                                    for (j = 0; j < p; ++j)
                                    {
                                        for (k = 0; k < n; ++k)
                                        {
                                            multiply[i][j] += first[i][k] * second[k][j];
                                        }
                                    }
                                }
                        
                                // Display the result matrix
                                printf("Product of entered matrices: \n");
                                for (i = 0; i < m; ++i)
                                {
                                    for (j = 0; j < p; ++j)
                                    {
                                        printf("%d  ", multiply[i][j]);
                                    }
                                    printf("\n");
                                }
                            }
                        
                            return 0;
                        }

                    void concatString(char str1[], char str2[], char str3[])

                        for (i = 0; str1[i] != '\0'; ++i) 
                            str3[i] = str1[i]; 
                        
                        for (j = 0; str2[j] != '\0'; ++j) 
                            str3[i + j] = str2[j]; 
                    
                        str3[i + j] = '\0'; 
                    }

                    {
